linux进程控制编程，通过一个相对复杂程序分析fork()函数

代码分析

  程序分析

#include <unistd.h>  
#include <stdio.h>  

int main(void)  
{  
	int i=0, k=0;  
	for(i=0;i<2;i++)
	{  
		pid_t fpid=fork();  
		if(fpid==0)
		{
			printf("son, %d, */n", ++k);
		}
		else
		{
			printf("father, %d, */n", ++k);
		}
	}  
	return 0;  
}
